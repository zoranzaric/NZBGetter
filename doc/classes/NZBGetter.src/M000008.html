  def download_nzbs(search_term)
    source = get_rss_url(search_term)
    content = ""
    open(source) do |s| content = s.read end
    rss = RSS::Parser.parse(content, false)
    rss.items.each do |item|
      puts "downloading " + item.title + "..."
      download_nzb(item)
    end
  end
